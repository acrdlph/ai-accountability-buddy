```python
import asyncio, os
from dotenv import load_dotenv
load_dotenv(dotenv_path=".env.local")
from habitify import fetch_journal, _format_target_date

api_key = os.getenv("HABITIFY_API_KEY")
tz_offset = int(os.getenv("TZ_OFFSET_HOURS", "1"))
target_date = _format_target_date(tz_offset)

habits = asyncio.run(fetch_journal(api_key, target_date))
for h in habits:
    goal = h.get("goal")
    print(f"  {h['name']} | status={h.get('status','?')} | goal={'yes: ' + str(goal['value']) + ' ' + goal['unit_type'] if goal else 'no'}")
print(f"\nTotal: {len(habits)} habits")
```
